- 使用中文标点符号

- 代码块保持原格式

  // 检查必需字段
  if (!userData.username || !userData.email) {
    throw new Error('缺少必需的用户信息');
  }

  // 返回验证结果
  return {
    isValid: true,
    message: '用户验证通过'
  };
}
